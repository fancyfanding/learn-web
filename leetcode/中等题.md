# [49. 字母异位词分组](https://leetcode.cn/problems/group-anagrams/)

/**
 * @param {string[]} strs
 * @return {string[][]}
 */

function strSort(str) {
    let chars = []
    for (let i = 0; i < str.length; i++) {
        chars.push(str[i])
    }
    chars.sort();
    let newStr = "";
    for (let i = 0; i < chars.length; i++) {
        newStr += chars[i]
    }
    return newStr;
}
var groupAnagrams = function (strs) {
    let newStrs = []
    for (let i = 0; i < strs.length; i++) {
        let obj = { op: i, value: "" }
        obj.value = strSort(strs[i]);
        newStrs[i] = obj;
    }

    newStrs = [...newStrs].sort((a, b) => a.value.localeCompare(b.value));
    let itemArr = [], ansArr = [];
    for (let i = 0; i < newStrs.length; i++) {
        itemArr.push(strs[newStrs[i].op])
        if (i === newStrs.length - 1 || newStrs[i].value !== newStrs[i + 1].value) {
            ansArr.push(itemArr)
            itemArr = []
        }
    }
    return ansArr
};

    newNums = new Set(nums)
    newNums = [...newNums]
    for (let i = 1; i < newNums.length; i++) {
        if (newNums[i] - newNums[i - 1] === 1) {

# [11. 盛最多水的容器](https://leetcode.cn/problems/container-with-most-water/)

```javascript
/**
 * @param {number[]} height
 * @return {number}
 */
var maxArea = function (height) {
    let left = 0, right = height.length - 1;
    let maxn = -1;
    while (left <= right) {
        maxn = Math.max(maxn, Math.min(height[left], height[right]) * (right - left));
        if (height[left] < height[right]) left++
        else right--;
    }
    return maxn
};
```

# [15. 三数之和](https://leetcode.cn/problems/3sum/)

```javascript
/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var threeSum = function (nums) {
    nums.sort((a, b) => a - b);

    let ans = [];
    for (let i = 0; i < nums.length - 2; i++) {
        if (nums[i] > 0) break;
        if (i > 0 && nums[i] === nums[i - 1]) {
            continue;
        }
        let left = i + 1, right = nums.length - 1;
        while (left < right) {
            if (nums[left] === nums[left - 1] && i != left - 1) {
                left++;
                continue;
            }
            if (nums[right] === nums[right + 1]) {
                right--;
                continue;
            }
            if (nums[i] + nums[left] + nums[right] === 0) {
                ans.push([nums[i], nums[left], nums[right]])
                left++, right--;
            }
            else if (nums[i] + nums[left] + nums[right] > 0) {
                right--;
            }
            else {
                left++;
            }
        }
    }
    return ans
};
```

    let a=[],maxn=0;

# [438. 找到字符串中所有字母异位词](https://leetcode.cn/problems/find-all-anagrams-in-a-string/)

```javascript
/**
 * @param {string} s
 * @param {string} p
 * @return {number[]}
 */
var findAnagrams = function(s, p) {
    let ans = [];
    const sLen = s.length;
    const pLen = p.length;
  
    if (sLen < pLen) return ans;
  
    const tp = new Array(26).fill(0); // 目标字符串 p 的字符计数
    const sp = new Array(26).fill(0); // 滑动窗口的字符计数
  
    // 1. 初始化目标计数和第一个窗口
    for (let i = 0; i < pLen; i++) {
        tp[p.charCodeAt(i) - 97]++;
        sp[s.charCodeAt(i) - 97]++;
    }
  
    // 2. 初始化匹配的字符数
    let matchCount = 0;
    for (let i = 0; i < 26; i++) {
        if (tp[i] === sp[i]) matchCount++;
    }
  
    // 3. 滑动窗口遍历
    for (let i = 0; i <= sLen - pLen; i++) {
        // 如果当前窗口完全匹配，记录起始索引
        if (matchCount === 26) {
            ans.push(i);
        }
    
        // 窗口右移：移除左侧字符
        const leftCharIdx = s.charCodeAt(i) - 97;
        if (sp[leftCharIdx] === tp[leftCharIdx]) matchCount--;
        sp[leftCharIdx]--;
        if (sp[leftCharIdx] === tp[leftCharIdx]) matchCount++;
    
        // 窗口右移：添加右侧字符（避免最后一次越界）
        if (i + pLen < sLen) {
            const rightCharIdx = s.charCodeAt(i + pLen) - 97;
            if (sp[rightCharIdx] === tp[rightCharIdx]) matchCount--;
            sp[rightCharIdx]++;
            if (sp[rightCharIdx] === tp[rightCharIdx]) matchCount++;
        }
    }
  
    return ans;
};
```
